% First-page title banner and author block.
\makeatletter
\renewcommand{\maketitle}{
  \begin{tikzpicture}[remember picture,overlay]
    \fill[bannercolor] (current page.north west) rectangle ([yshift=-6.7cm]current page.north east);
    \node[anchor=west] at ([xshift=0.30cm,yshift=-1.55cm]current page.north west) {\climatelogo};
    \node[
      name=titleblock,
      anchor=west,
      align=left,
      text width=\dimexpr\paperwidth-3.9cm\relax
    ] at ([xshift=2.55cm,yshift=-1.65cm]current page.north west) {%
      {\color{white}\displayfont\bfseries\parbox[t]{\dimexpr\paperwidth-3.9cm\relax}{\raggedright\hyphenpenalty=10000\exhyphenpenalty=10000\fontsize{24}{26}\selectfont\MakeUppercase{\@title}}}
    };
    \node[
      anchor=north west,
      align=left,
      text width=\dimexpr\paperwidth-3.9cm\relax
    ] at ([yshift=-4mm]titleblock.south west) {%
      {\color{white}\displayfont\bfseries\parbox[t]{\dimexpr\paperwidth-3.9cm\relax}{\raggedright\hyphenpenalty=10000\exhyphenpenalty=10000\fontsize{16}{19}\selectfont\MakeUppercase{\@subtitle}}}
    };
    \node[
      anchor=west,
      align=left,
      text width=\dimexpr\paperwidth-5.0cm\relax
    ] at ([xshift=2.55cm,yshift=-5.85cm]current page.north west) {%
      {\color{white}\displayfont\fontsize{14}{17}\selectfont \@author}
    };
  \end{tikzpicture}
  \mbox{}
  \vspace*{4.6cm}
}
\makeatother
